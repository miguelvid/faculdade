CREATE TABLE LOCAL (
    PKLOCAL INT PRIMARY KEY,
    DSLOCAL CHAR(30)
);

CREATE TABLE Professor (
    PKPROFESSOR INT PRIMARY KEY,
    DSNOME CHAR(30),
    NMTELEFONE CHAR(30)
);

CREATE TABLE Aluno (
    PKALUNO INT PRIMARY KEY,
    DSNOME CHAR(30),
    NMTELEFONE CHAR(30)
);

CREATE TABLE Curso (
    PKCURSO INT PRIMARY KEY,
    DSNOME CHAR(30),
    NMPRECO CHAR(30),
    DSDURACAO CHAR(30),
    FKPROFESSOR INT,
    FOREIGN KEY (FKPROFESSOR) REFERENCES Professor(PKPROFESSOR)
);

CREATE TABLE Matricula (
    PKMATRICULA INT PRIMARY KEY,
    FKALUNO INT,
    FKCURSO INT,
    FOREIGN KEY (FKALUNO) REFERENCES Aluno(PKALUNO),
    FOREIGN KEY (FKCURSO) REFERENCES Curso(PKCURSO)
);
